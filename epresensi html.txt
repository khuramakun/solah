    function initDarkMode() {
      const savedTheme = localStorage.getItem('pondok-theme');
      const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;

        document.documentElement.classList.add('dark');
      } else {
        document.documentElement.classList.remove('dark');
      }
      updateDarkModeButton();

    function playBeepSound(isSuccess) {
      try {
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        const oscillator = audioCtx.createOscillator();
        const gainNode = audioCtx.createGain();
        oscillator.connect(gainNode);
        gainNode.connect(audioCtx.destination);
        
        if (isSuccess) {
          oscillator.frequency.setValueAtTime(880, audioCtx.currentTime); 
          oscillator.type = "sine";
          gainNode.gain.setValueAtTime(0.1, audioCtx.currentTime);
          oscillator.start();
          oscillator.stop(audioCtx.currentTime + 0.15);
        } else {
          oscillator.frequency.setValueAtTime(220, audioCtx.currentTime); 
          oscillator.type = "sawtooth";
          gainNode.gain.setValueAtTime(0.15, audioCtx.currentTime);
          oscillator.start();
          oscillator.stop(audioCtx.currentTime + 0.35);
        }
      } catch (e) {
        console.warn("Audio Context blocked or not supported", e);
      }
    }

    function generateBarcodeSVG(text) {
      const CODE39_MAP = {
        '0': '101001101101', '1': '110100101011', '2': '101100101011', '3': '110110010101',
        '4': '101001101011', '5': '110100110101', '6': '101100110101', '7': '101001011011',
        '8': '110100101101', '9': '101100101101', 'A': '110101001011', 'B': '101101001011',
        'C': '110110100101', 'D': '101011001011', 'E': '110101100101', 'F': '101101100101',
        'G': '101010011011', 'H': '110101001101', 'I': '101101001101', 'J': '101011001101',
        'K': '110101010011', 'L': '101101010011', 'M': '110110101001', 'N': '101011010011',
        'O': '110101101001', 'P': '101101101001', 'Q': '101010110011', 'R': '110101011001',
        'S': '101101011001', 'T': '101011011001', 'U': '110010101011', 'V': '100110101011',
        'W': '110011010101', 'X': '100101101011', 'Y': '110010110101', 'Z': '100110110101',
        '-': '100101011011', '.': '110010101101', ' ': '100110101101', '*': '100101101101'
      };

      const rawStr = `*${text.toUpperCase()}*`;
      let fullBinaryPattern = "";
      
      for (let i = 0; i < rawStr.length; i++) {
        const char = rawStr[i];
        if (CODE39_MAP[char]) {
          fullBinaryPattern += CODE39_MAP[char] + "0"; // Spasi pemisah
        }
      }

      const rectWidth = 2;
      const height = 65;
      let svgRects = "";
      let currentX = 0;

      for (let j = 0; j < fullBinaryPattern.length; j++) {
        if (fullBinaryPattern[j] === '1') {
          svgRects += `<rect x="${currentX}" y="5" width="${rectWidth}" height="${height}" fill="black" />`;
        }
        currentX += rectWidth;
      }

      return `<svg width="100%" height="80" viewBox="0 0 ${currentX} 75" preserveAspectRatio="none" xmlns="http://www.w3.org/2000/svg" style="color: #000000 !important;">${svgRects}</svg>`;
    }

    // MEMAKSA KONEKSI KAMERA DAN MEMASTIKAN BROWSER MENYETUJUI IZIN HARDWARE
    async function startCameraScanner() {
      if (typeof Html5Qrcode === 'undefined') {
        showToast("Sedang memuat pustaka scanner kamera. Harap tunggu...", "warning");
        return;
      }
      stopCameraScanner();

      const readerElement = document.getElementById("reader-container");
      if (!readerElement) return;

      // Alur Paksa Koneksi: Meminta izin aktif secara eksplisit ke browser
      try {
        const testStream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } });
        testStream.getTracks().forEach(track => track.stop()); // Matikan kembali setelah disetujui agar html5-qrcode bisa menduduki stream utama
        document.getElementById("camera-error-msg").classList.add("hidden");
      } catch (err) {
        console.error("Izin akses kamera diblokir atau gagal:", err);
        document.getElementById("camera-error-msg").classList.remove("hidden");
        showToast("Gagal menyambungkan kamera. Harap setujui izin perangkat!", "error");
        return;
      }

      html5QrcodeScanner = new Html5Qrcode("reader-container");
      const config = { 
        fps: 15, 
        qrbox: { width: 250, height: 250 },
        aspectRatio: 1.777778
      };

      html5QrcodeScanner.start(
        { facingMode: "environment" }, 
        config,
        (decodedText) => {
          if (scanCooldown) return;
          scanCooldown = true;
          
          processVerifiedBarcode(decodedText);
          
          setTimeout(() => {
            scanCooldown = false;
          }, 3500);
        },
        (errorMessage) => {
          // Callback debugging
        }
      ).catch(err => {
        console.error("Gagal memulai modul kamera html5-qrcode:", err);
        const errAlert = document.getElementById("camera-error-msg");
        if (errAlert) errAlert.classList.remove("hidden");
      });
    }

    function stopCameraScanner() {
      if (html5QrcodeScanner) {
        try {
          html5QrcodeScanner.stop().then(() => {
            console.log("Kamera pemindai dihentikan secara aman.");
          }).catch(err => {
            // Abaikan penutupan jika tidak aktif
          });
        } catch (e) {
          // Exception silencer
        }
      }
    }

    function simulateBarcodeScan(barcode) {
      processVerifiedBarcode(barcode);
    }

    async function processVerifiedBarcode(barcode) {
      const res = await fetchAPI("scanStudentBarcode", { 
        barcode: barcode, 
        operatorName: currentSession.name 
      });
      
      if (res.success) {
        playBeepSound(true);
        showToast(res.message, "success");
        renderPage("scanner"); 
      } else {
        playBeepSound(false);
        showToast(res.message, "warning");
      }
    }

    function filterStudentTable() {
      const q = document.getElementById("student-search-bar").value.toLowerCase();
      document.querySelectorAll(".student-row").forEach(row => {

        const r_nis = row.querySelector(".target-nis").innerText.toLowerCase();
        if (name.includes(q) || r_nis.includes(q)) {

        } else {
          row.classList.add("hidden");
        }
      });
    }

    function updateLaporanFilter(key, value) {
      laporanFilters[key] = value;
      renderLaporanTableRows();
    }

    function renderLaporanTableRows() {
      const tbody = document.getElementById("laporan-table-body");
      const countLabel = document.getElementById("laporan-log-count");
      if (!tbody) return;

      const filtered = db.attendance.filter(record => {
        if (laporanFilters.date && record.date !== laporanFilters.date) {
          return false;
        }

        if (laporanFilters.classId) {
          const student = db.students.find(s => s.id === record.studentId);
          if (!student || student.classId !== laporanFilters.classId) {
            return false;
          }
        }

        if (laporanFilters.status && record.status !== laporanFilters.status) {
          return false;
        }

        return true;
      });

      if (countLabel) {
        countLabel.innerText = `${filtered.length} Entri Log`;
      }

      if (filtered.length === 0) {
        tbody.innerHTML = `
          <tr>
            <td colspan="6" class="py-12 text-center text-slate-400 dark:text-slate-500">
              Tidak ada log kehadiran santri yang cocok dengan filter.
            </td>
          </tr>
        `;
        return;
      }

      tbody.innerHTML = filtered.map(a => `
        <tr class="hover:bg-slate-50/50 dark:hover:bg-slate-800/30 transition">
          <td class="px-5 py-3.5">
            <p class="font-bold text-slate-850 dark:text-white">${a.date}</p>
            <p class="text-[10px] text-slate-400 dark:text-slate-500 font-mono">${a.time}</p>
          </td>
          <td class="px-5 py-3.5 font-bold text-slate-800 dark:text-white">${a.name}</td>
          <td class="px-5 py-3.5 text-slate-500 dark:text-slate-400 font-medium">${a.className}</td>
          <td class="px-5 py-3.5 text-slate-400 dark:text-slate-500 italic">${a.scanner}</td>
          <td class="px-5 py-3.5">
            <span class="px-2.5 py-1 rounded-full text-[9px] font-bold uppercase ${
              a.status === 'Hadir' ? 'bg-blue-100 dark:bg-blue-500/20 text-blue-800 dark:text-blue-300' :
              a.status === 'Terlambat' ? 'bg-indigo-100 dark:bg-indigo-500/20 text-indigo-800 dark:text-indigo-300' :
              a.status === 'Sakit' ? 'bg-amber-100 dark:bg-amber-500/20 text-amber-800 dark:text-amber-300' : 
              a.status === 'Izin' ? 'bg-cyan-100 dark:bg-cyan-500/20 text-cyan-800 dark:text-cyan-300' : 'bg-rose-100 dark:bg-rose-500/20 text-rose-800 dark:text-rose-300'
            }">${a.status}</span>
          </td>
          <td class="px-5 py-3.5 text-right no-print">
            ${currentSession.role === 'Administrasi' ? `
              <button onclick="editAttendanceStatus('${a.id}')" class="text-blue-600 dark:text-cyan-400 hover:text-blue-800 dark:hover:text-cyan-300 font-bold text-[10px] uppercase">
                Ganti Status
              </button>
            ` : `<span class="text-slate-300 dark:text-slate-700">-</span>`}
          </td>
        </tr>
      `).join("");
    }

    function editAttendanceStatus(id) {
      const record = db.attendance.find(a => a.id === id);
      if (!record) return;

      const modalHTML = `
        <div class="p-5 border-b border-slate-100 dark:border-slate-800 flex items-center justify-between">
          <h3 class="font-bold text-slate-800 dark:text-white text-sm">Ganti Status Absensi: ${record.name}</h3>
          <button onclick="closeModal()" class="text-slate-400 hover:text-slate-600 dark:hover:text-slate-200"><i class="fa-solid fa-xmark"></i></button>
        </div>
        <div class="p-5 space-y-4 bg-white dark:bg-slate-900 rounded-b-2xl">
          <p class="text-xs text-slate-500 dark:text-slate-400">Ubah kehadiran manual santri untuk tanggal <strong>${record.date}</strong>:</p>
          <div class="grid grid-cols-2 sm:grid-cols-4 gap-2">
            <button onclick="updateStatus('${record.id}', 'Hadir')" class="py-2 rounded-lg font-bold text-xs bg-blue-50 dark:bg-blue-500/10 text-blue-700 dark:text-cyan-300 border border-blue-200 dark:border-blue-500/30">Hadir</button>
            <button onclick="updateStatus('${record.id}', 'Terlambat')" class="py-2 rounded-lg font-bold text-xs bg-indigo-50 dark:bg-indigo-500/10 text-indigo-700 dark:text-indigo-300 border border-indigo-200 dark:border-indigo-500/30">Terlambat</button>
            <button onclick="updateStatus('${record.id}', 'Sakit')" class="py-2 rounded-lg font-bold text-xs bg-amber-50 dark:bg-amber-500/10 text-amber-700 dark:text-amber-300 border border-amber-200 dark:border-amber-500/30">Sakit</button>
            <button onclick="updateStatus('${record.id}', 'Izin')" class="py-2 rounded-lg font-bold text-xs bg-cyan-50 dark:bg-cyan-500/10 text-cyan-700 dark:text-cyan-300 border border-cyan-200 dark:border-cyan-500/30">Izin</button>
          </div>
        </div>
      `;

      openModal(modalHTML);
    }

    async function updateStatus(id, newStatus) {
      const res = await fetchAPI("updateAttendanceStatus", { id: id, status: newStatus });
      if (res.success) {
        showToast(res.message, "success");
        closeModal();
        
        if (!isGasEnvironment) {
          const target = db.attendance.find(a => a.id === id);
          if (target) target.status = newStatus;
        } else {
          const stateRes = await fetchAPI("getInitialState");
          if (stateRes.success) db = stateRes.data;
        }

        renderLaporanTableRows();
      } else {
        showToast(res.message, "error");
      }
    }
